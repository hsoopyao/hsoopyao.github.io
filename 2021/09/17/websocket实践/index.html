浅谈WebSocket

WebSocket是在HTML5基础上单个TCP连接上进行全双工通讯的协议，只要浏览器和服务器进行一次握手，就可以建立一条快速通道，两者就可以实现数据互传了，就是打破了传统的http协议的无状态传输（只能浏览器请求，服务端响应），websocket全双工通讯，就是浏览器和服务器进行一次握手，浏览器可以随时给服务器发送信息，服务器也可以随时主动发送信息给浏览器了。

stomp协议

我们先来了解stomp是什么。

STOMP即Simple (or Streaming) Text Orientated Messaging Protocol，简单(流)文本定向消息协议，它提供了一个可互操作的连接格式，允许STOMP客户端与任意STOMP消息代理（Broker）进行交互。STOMP协议由于设计简单，易于开发客户端，因此在多种语言和多种平台上得到广泛地应用。

SpringBoot + WebSocket + stomp

首先的首先导入依赖

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-websocket</artifactId>
</dependency>

WebSocket配置类

  • addEndpoint：设置连接websocket服务器的端点
  • enableSimpleBroker：用户可以通过订阅这两个消息代理来接受消息(可以把这两个消息代理想象成两条管道)
  • setApplicationDestinationPrefixes：以/app为前缀的会发送到服务器中
  • setUserDestinationPrefix：以/user为前缀的会发送到对应的客户端中
@Configuration
//开启使用stomp协议来传输信息
@EnableWebSocketMessageBroker
public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {

    @Override
    //注册stomp协议的端点(endpoint)
    public void registerStompEndpoints(StompEndpointRegistry registry) {
        //注册一个stomp协议的访问端点，并指定使用sockJS协议以及支持跨域
        //通过localhost:port/websocket 进行连接websocket
        registry.addEndpoint("/websocket")
                .setAllowedOrigins("*")
                .withSockJS();
    }

    @Override
    //配置消息代理
    public void configureMessageBroker(MessageBrokerRegistry registry) {
        //客户端订阅路径前缀（基于内存的STOMP消息代理）
        registry.enableSimpleBroker("/topic","/all");
        //用户发送消息到服务器的前缀
        registry.setApplicationDestinationPrefixes("/app");
        //客户端对客户端之间的通信的前缀(即一对一)
        registry.setUserDestinationPrefix("/user");
    }
}

题外话：WebSocket还可以配置拦截器

  • 实现ChannelInterceptor接口
  • 这里集成了SpringEvent事件监听机制
  • preSend：是在发送前的处理
  • postSend：是在发送时的处理
    • 通过MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);来获取消息头的信息，可以获取到sessionId，command…
/**
 * websocketChannel拦截器
 * @author PeanuTuT
 */
@Slf4j
@Component
public class WebSocketChannelInterceptor implements ChannelInterceptor {

    //这是SpringEvent的发送器
    private final ApplicationEventPublisher publisher;

    //通过构造器的方式注入 这里其实省略了@Autowired这个注解
    public WebSocketChannelInterceptor(ApplicationEventPublisher publisher) {
        this.publisher = publisher;
    }

    @Override
    public Message<?> preSend(Message<?> message, MessageChannel channel) {
        //获取信息头
        StompHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);
        if (accessor == null){
            return null;
        }
        String sessionId = accessor.getSessionId();
        if (StompCommand.CONNECT.equals(accessor.getCommand())){
            log.info("realtime client {} connection ...",sessionId);
        } else if (StompCommand.DISCONNECT.equals(accessor.getCommand())){
            log.info("realtime client {} disconnection ...",sessionId);
        }
        return message;
    }

    @Override
    public void postSend(Message<?> message, MessageChannel channel, boolean sent) {
        StompHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);
        if (accessor == null){
            return;
        }
        String sessionId = accessor.getSessionId();
        if (StompCommand.SUBSCRIBE.equals(accessor.getCommand())){
            String destination = accessor.getDestination();
            log.info("realtime client {} subscribe {} ...",sessionId,destination);
            publisher.publishEvent(ConnectEvent.getInstance(this,sessionId,destination));
        }
    }
}

还需要在WebSocket配置类中配置拦截器

@Configuration
@EnableWebSocketMessageBroker
public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {

    private final WebSocketChannelInterceptor channelInterceptor;

    public WebSocketConfig(WebSocketChannelInterceptor channelInterceptor) {
        this.channelInterceptor = channelInterceptor;
    }

    @Override
    public void configureClientInboundChannel(ChannelRegistration registration) {
        registration.interceptors(channelInterceptor);
    }
}

Controller层

@Slf4j
@Controller
public class WebSocketController {

    private final SimpMessagingTemplate messagingTemplate;

    public WebSocketController(SimpMessagingTemplate messagingTemplate) {
        this.messagingTemplate = messagingTemplate;
    }

    @CrossOrigin
    //类似与@RequestMapping 对应的请求地址会映射到这个方法上来
    @MessageMapping(value = "/room/{roomId}")
    public void broadcast(@DestinationVariable("roomId") String roomId, @Payload RequestMessage requestMessage){
        //发送的地址，发送到以/topic为前缀的通道里，只要有用户订阅这个通道就会收到信息
        String destination = "/topic/room/" + HtmlUtils.htmlEscape(roomId);
        //messagingTemplate相当于@SendTo注解
        messagingTemplate.convertAndSend(destination,requestMessage);
    }
}

单元测试一波

@Slf4j
public class WebSocketDemoTest {

    private int port = 8888;

    private WebSocketStompClient stompClient;

    private final WebSocketHttpHeaders headers = new WebSocketHttpHeaders();

    //构建消息
    RequestMessage testMessage = RequestMessage.builder().sender("11111111111").content("这是一条弹幕").type("bullet").build();
    String         line        = "\n---------------------------------------------------------------------\n";

    //每个实例之前执行
    @BeforeEach
    public void setup(){
        //transport:负载
        List<Transport>transports = new ArrayList<>();
        WebSocketContainer container = ContainerProvider.getWebSocketContainer();
        container.setDefaultMaxBinaryMessageBufferSize(1024 * 1024);
        container.setDefaultMaxTextMessageBufferSize(1024 * 1024);
        transports.add(new WebSocketTransport(new StandardWebSocketClient(container)));
        SockJsClient sockJsClient = new SockJsClient(transports);
        this.stompClient = new WebSocketStompClient(sockJsClient);
        this.stompClient.setMessageConverter(new MappingJackson2MessageConverter());
    }

    @Test
    public void test() throws InterruptedException {
        final CountDownLatch latch = new CountDownLatch(1);
        final AtomicReference<Throwable> failure = new AtomicReference<>();

        StompSessionHandler handler = new TestSessionHandler(failure) {

            @Override
            public void afterConnected(final StompSession session, StompHeaders connectedHeaders) {
                // 订阅/topic/room/1
                session.subscribe("/topic/room/1", new StompFrameHandler() {
                    @Override
                    public Type getPayloadType(StompHeaders headers) {
                        return RequestMessage.class;
                    }

                    @Override
                    public void handleFrame(StompHeaders headers, Object payload) {
                        //接受消息
                        RequestMessage message = (RequestMessage) payload;
                        try {
                            log.info("{}receive: {}{}", line, message, line);
                        } catch (Throwable t) {
                            failure.set(t);
                        } finally {
                            session.disconnect();
                            latch.countDown();
                        }
                    }
                });

                try {
                    // 发布实时消息
                    session.send("/app/room/1", testMessage);
                } catch (Throwable t) {
                    failure.set(t);
                    latch.countDown();
                }
            }
        };
		//连接websocket 开启双通道
        this.stompClient.connect("ws://localhost:{port}/websocket", this.headers, handler, this.port);

        if (latch.await(60, TimeUnit.SECONDS)) {
            if (failure.get() != null) {
                throw new AssertionError("", failure.get());
            }
        } else {
            fail("Greeting not received");
        }
    }

    private class TestSessionHandler extends StompSessionHandlerAdapter {

        private final AtomicReference<Throwable> failure;

        public TestSessionHandler(AtomicReference<Throwable> failure) {
            this.failure = failure;
        }

        @Override
        public void handleFrame(StompHeaders headers, Object payload) {
            this.failure.set(new Exception(headers.toString()));
        }

        @Override
        public void handleException(StompSession s, StompCommand c, StompHeaders h, byte[] p, Throwable ex) {
            log.error("", ex);
            this.failure.set(ex);
        }

        @Override
        public void handleTransportError(StompSession session, Throwable ex) {
            this.failure.set(ex);
        }
    }
}
